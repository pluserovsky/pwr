#include <cstdlib>
#include <iostream>

int main()
{
    int m=40,i,k;
    char ciag[m];
    char z;
    cout<<"Podaj ciag znakow (max znakow 39):"<<endl;
    cin.getline(ciag,m);                //wczytanie ciagu znakow wraz ze spacjami
    while((int)ciag[0]==32)             //sprawdzenie czy ciag zawiera znaki spacji na poczatku
    {i=0;                               
    for(k=1;k<m;k++)                    
    {ciag[i]=ciag[k];                   //jezeli znaki spacji wystepuja to 
    i++;                                //tekst jest przesuwany o jedne znak w lewo
    }}                                  
    i=0;
    while((int)ciag[i]!=0)              //policzenie ile znkaow zawiera ciag
    i++;
    for(i--;(int)ciag[i]==32;i--)       //sprawdzanie czy ostatni znak jest spacja
    ciag[i]=0;                          //jesli tak zostaje on usuniety
    cout<<"|"<<ciag<<"|"<<endl;         //zwrocenie tekstu bez spacji | zostaly u¿yte ¿eby
